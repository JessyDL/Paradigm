137  static psl::string to_unix(psl::string_view path)
138  {
139  psl::string dir;
140  while(path.size() > 0 && (path[0] == '\"' || path[0] == '\''))
141  {
142  path = path.substr(1);
143  }
144 
145  while(path.size() > 0 && (path[path.size() - 1] == '\"' || path[path.size() - 1] == '\''))
146  {
147  path = path.substr(0, path.size() - 1);
148  }
149 
150  if(auto find = path.find(":\\\\"); find == 1u)
151  {
152  psl::char_t drive = utility::string::to_lower(path.substr(0u, 1u))[0];
153  dir = "/mnt/" + psl::string(1, drive) + seperator + path.substr(4u);
154  }
155  else if(find = path.find(":\\"); find == 1u)
156  {
157  psl::char_t drive = utility::string::to_lower(path.substr(0u, 1u))[0];
158  dir = "/mnt/" + psl::string(1, drive) + seperator + path.substr(3u);
159  }
160  else if(find = path.find(":/"); find == 1u)
161  {
162  psl::char_t drive = utility::string::to_lower(path.substr(0u, 1u))[0];
163  dir = "/mnt/" + psl::string(1, drive) + seperator + path.substr(3u);
164  }
165  else
166  {
167  dir = path;
168  }
169 
170  auto find = dir.find('\\');
171  while(find < dir.size())
172  {
173  dir.replace(find, 1, "/");
174  find = dir.find('\\');
175  }
176 
177  return dir;
178  }
