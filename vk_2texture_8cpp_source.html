182 void texture::load_2D()
183 {
184  gli::texture2d* m_Texture2DData = (gli::texture2d*)m_TextureData;
185  if(m_Texture2DData->empty())
186  {
187  LOG_ERROR("Empty texture");
188  debug_break();
189  }
190 
191  if(m_Meta->width() != (uint32_t)(*m_Texture2DData)[0].extent().x)
192  m_Meta->width((uint32_t)(*m_Texture2DData)[0].extent().x);
193 
194  if(m_Meta->height() != (uint32_t)(*m_Texture2DData)[0].extent().y)
195  m_Meta->height((uint32_t)(*m_Texture2DData)[0].extent().y);
196 
197  if(m_Meta->depth() != (uint32_t)(*m_Texture2DData)[0].extent().z)
198  m_Meta->depth((uint32_t)(*m_Texture2DData)[0].extent().z);
199 
200  m_MipLevels = (uint32_t)m_Texture2DData->levels();
201  m_Meta->mip_levels(m_MipLevels);
202  vk::FormatProperties formatProperties;
203 
204  // todo we can map gli::format to the vulkan mappings.
205  if(m_Meta->format() == vk::Format::eUndefined)
206  {
207  LOG_ERROR("Undefined format property in: ", m_Meta->ID().to_string());
208  }
209  m_Context->physical_device().getFormatProperties(m_Meta->format(), &formatProperties);
210 
211  // Only use linear tiling if requested (and supported by the device)
212  // Support for linear tiling is mostly limited, so prefer to use
213  // optimal tiling instead
214  // On most implementations linear tiling will only support a very
215  // limited amount of formats and features (mip maps, cubemaps, arrays, etc.)
216  vk::Bool32 useStaging = m_StagingBuffer;
217 
218  // Only use linear tiling if forced
219  if(!useStaging)
220  {
221  // Don't use linear if format is not supported for (linear) shader sampling
222  useStaging = !(formatProperties.linearTilingFeatures & vk::FormatFeatureFlagBits::eSampledImage);
223  }
224 
225  // todo write the version without staging
226  useStaging = true;
227 
228  if(useStaging)
229  {
230  auto stagingBuffer = m_StagingBuffer;
231  if(!m_StagingBuffer)
232  {
233  core::ivk::log->warn("missing a staging buffer in ivk::texture, will create one dynamically, but this is inneficient");
234  auto tempBuffer = core::resource::create<core::data::buffer>(m_Cache);
235  tempBuffer.load(vk::BufferUsageFlagBits::eTransferSrc,
236  vk::MemoryPropertyFlagBits::eHostVisible | vk::MemoryPropertyFlagBits::eHostCoherent,
237  memory::region{m_Texture2DData->size() + 1024, 4, new memory::default_allocator(false)});
238  stagingBuffer = create<gfx::buffer>(m_Cache);
239  stagingBuffer.load(m_Context, tempBuffer);
240  }
241  if(!stagingBuffer)
242  {
243  core::ivk::log->error("could not create a staging buffer in ivk::texture");
244  return;
245  }
246  memory::segment segment;
247  if(auto segmentOpt = stagingBuffer->reserve((vk::DeviceSize)m_Texture2DData->size()); segmentOpt)
248  {
249  segment = segmentOpt.value();
250  buffer::commit_instruction instr;
251  instr.segment = segment;
252  instr.size = (vk::DeviceSize)m_Texture2DData->size();
253  instr.source = (std::uintptr_t)m_Texture2DData->data();
254  if(!stagingBuffer->commit({instr}))
255  {
256  core::ivk::log->error("could not commit an ivk::texture in a staging buffer");
257  }
258  }
259  else
260  {
261  core::ivk::log->error("could not allocate a segment in the staging buffer for an ivk::texture");
262  return;
263  }
264  //stagingBuffer->map(m_Texture2DData->data(), (vk::DeviceSize)m_Texture2DData->size(), 0);
265 
266  // Setup buffer copy regions for each mip level
267  std::vector<vk::BufferImageCopy> bufferCopyRegions;
268  uintptr_t offset = segment.range().begin;
269 
270  for(uint32_t i = 0; i < m_MipLevels; i++)
271  {
272  vk::BufferImageCopy bufferCopyRegion;
273  bufferCopyRegion.imageSubresource.aspectMask = vk::ImageAspectFlagBits::eColor;
274  bufferCopyRegion.imageSubresource.mipLevel = i;
275  bufferCopyRegion.imageSubresource.baseArrayLayer = 0;
276  bufferCopyRegion.imageSubresource.layerCount = m_Meta->layers();
277  bufferCopyRegion.imageExtent.width = (uint32_t)(*m_Texture2DData)[i].extent().x;
278  bufferCopyRegion.imageExtent.height = (uint32_t)(*m_Texture2DData)[i].extent().y;
279  bufferCopyRegion.imageExtent.depth = 1;
280  bufferCopyRegion.bufferOffset = offset;
281 
282  bufferCopyRegions.push_back(bufferCopyRegion);
283 
284  offset += (*m_Texture2DData)[i].size();
285  }
286

288  vk::ImageCreateInfo imageCreateInfo;
289  imageCreateInfo.pNext = NULL;
290  imageCreateInfo.imageType = (vk::ImageType)m_Meta->image_type();
291  imageCreateInfo.format = m_Meta->format();
292  imageCreateInfo.mipLevels = m_MipLevels;
293  imageCreateInfo.arrayLayers = 1;

297  imageCreateInfo.sharingMode = vk::SharingMode::eExclusive;
298  imageCreateInfo.initialLayout = vk::ImageLayout::eUndefined;
299  imageCreateInfo.extent = vk::Extent3D{m_Meta->width(), m_Meta->height(), m_Meta->depth()};
300  imageCreateInfo.usage = m_Meta->usage();
